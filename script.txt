-- ══════════════════════════════════════════════
-- ИСПРАВЛЕННАЯ функция показа реального TradeGUI
-- Основано на реальной структуре из .rbxlx файла:
-- TradeGUI → Container → Items (Visible=false!) → Tabs, Trade, BottomBar
-- ══════════════════════════════════════════════
local function showRealTradeGUI(partnerName)
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")

    -- Ждём появления TradeGUI до 5 секунд
    local tradeGui = nil
    for i = 1, 50 do
        tradeGui = playerGui:FindFirstChild("TradeGUI_Phone")
                or playerGui:FindFirstChild("TradeGUI")
        if tradeGui then break end
        task.wait(0.1)
    end

    if not tradeGui then
        return warn("[MM2 TRADE] TradeGUI не найден в PlayerGui")
    end

    print("[MM2 TRADE] Найден GUI: " .. tradeGui.Name)

    -- 1. Включаем ScreenGui
    tradeGui.Enabled = true

    -- 2. BG — фоновое изображение (по умолчанию Visible=false)
    local bg = tradeGui:FindFirstChild("BG")
    if bg then bg.Visible = true end

    -- 3. Главный Container
    local container = tradeGui:FindFirstChild("Container")
    if not container then
        return warn("[MM2 TRADE] Container не найден")
    end

    -- 4. ГЛАВНОЕ: Items по умолчанию Visible=false — включаем его
    local items = container:FindFirstChild("Items")
    if items then
        items.Visible = true
        print("[MM2 TRADE] ✓ Items.Visible = true")

        -- 4a. Background (рамка фона)
        local background = items:FindFirstChild("Background")
        if background then background.Visible = true end

        -- 4b. Tabs — вкладки (Weapon, Pets, Search)
        local tabs = items:FindFirstChild("Tabs")
        if tabs then
            tabs.Visible = true

            -- Активируем вкладку Weapon (включаем ViewBorder, скрываем DontView)
            local weaponTab = tabs:FindFirstChild("Weapon")
            if weaponTab then
                local viewBorder = weaponTab:FindFirstChild("ViewBorder")
                local dontView   = weaponTab:FindFirstChild("DontView")
                if viewBorder then viewBorder.Visible = true end
                if dontView   then dontView.Visible   = false end
            end

            -- Pets вкладка — скрываем ViewBorder
            local petsTab = tabs:FindFirstChild("Pets")
            if petsTab then
                local viewBorder = petsTab:FindFirstChild("ViewBorder")
                local dontView   = petsTab:FindFirstChild("DontView")
                if viewBorder then viewBorder.Visible = false end
                if dontView   then dontView.Visible   = true end
            end
        end

        -- 4c. Main → Weapons ScrollingFrame (активная вкладка)
        local main = items:FindFirstChild("Main")
        if main then
            main.Visible = true
            local weaponsScroll = main:FindFirstChild("Weapons")
            local petsScroll    = main:FindFirstChild("Pets")
            if weaponsScroll then weaponsScroll.Visible = true end
            if petsScroll    then petsScroll.Visible    = false end
        end

        -- 4d. Trade панели (левая и правая)
        for _, child in pairs(items:GetChildren()) do
            if child.Name == "Trade" then
                child.Visible = true
            end
        end

        -- 4e. BottomBar — кнопки Accept / Decline
        local bottomBar = items:FindFirstChild("BottomBar")
        if bottomBar then
            bottomBar.Visible = true

            -- Подключаем нативную кнопку Accept к нашему autoConfirmTrade
            for _, acceptFrame in pairs(bottomBar:GetChildren()) do
                if acceptFrame:IsA("Frame") then
                    local btn = acceptFrame:FindFirstChildWhichIsA("TextButton")
                    if btn then
                        if btn.Text == "Accept" then
                            btn.MouseButton1Click:Connect(function()
                                print("[MM2 TRADE] Нативный Accept нажат")
                                autoConfirmTrade()
                            end)
                        elseif btn.Text == "Decline" then
                            btn.MouseButton1Click:Connect(function()
                                print("[MM2 TRADE] Decline нажат — скрываем GUI")
                                items.Visible = false
                                if bg then bg.Visible = false end
                                mockState.active = false
                            end)
                        end
                    end
                end
            end
        end
    else
        warn("[MM2 TRADE] Items не найден внутри Container")
    end

    mockState.active = true
    mockState.tradePartner = partnerName
    print("[MM2 TRADE] ✓ TradeGUI открыт для: " .. tostring(partnerName))
end
